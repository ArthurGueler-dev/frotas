
    // GET /api/ituran/full-report - Buscar relat√≥rio completo do Ituran
    app.get('/api/ituran/full-report', async (req, res) => {
        try {
            const { plate, start, end } = req.query;

            if (!plate || !start || !end) {
                return res.status(400).json({
                    success: false,
                    error: 'Par√¢metros obrigat√≥rios: plate, start, end'
                });
            }

            console.log(`üì° Buscando relat√≥rio Ituran: ${plate} (${start} at√© ${end})`);

            // Usar o ituranService que j√° est√° dispon√≠vel no server.js
            const report = await ituranService.getFullReport(plate, start, end);

            res.json(report);

        } catch (error) {
            console.error('‚ùå Erro ao buscar relat√≥rio Ituran:', error);
            res.status(500).json({
                success: false,
                error: error.message || 'Erro ao buscar relat√≥rio'
            });
        }
    });
